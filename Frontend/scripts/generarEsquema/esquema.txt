export function flattenTree(node, path = []) {
  let result = []; // Lista donde se acumulan los nodos
  const currentPath = [...path, node.nombre]; // Path actual (padres + este nodo)

  // Agrega el nodo actual con su path
  result.push({ ...node, path: currentPath });

  // Si tiene hijos, aplana cada uno y los concatena al resultado
  if (node.children && node.children.length > 0) {
    for (const child of node.children) {
      result = result.concat(flattenTree(child, currentPath));
    }
  }

  return result; // Devuelve la lista final
}

  const [position, setPosition] = useState({ x: 0, y: 0 });

  // Estado para controlar si se debe girar el cursor
  const [rotating, setRotating] = useState(false);

  // Estado para determinar si el cursor estÃ¡ sobre un botÃ³n o link
  const [hovered, setHovered] = useState(false);

  // Referencia al timeout que detiene la animaciÃ³n de giro si no hay movimiento
  const stopTimeout = useRef(null);

      // Limpia cualquier timeout anterior

      // Programa detener la rotaciÃ³n si no se mueve por 100ms

      }, 100);
    };

    // Si el mouse entra en un link o botÃ³n, activa el "hover"
    const handleMouseOver = (e) => {
      if (e.target.closest("a, button")) {
        setHovered(true);
      }
    };

    // Si el mouse sale del link o botÃ³n, desactiva el "hover"

      if (e.target.closest("a, button")) {
        setHovered(false);
      }
    };

    // Asigna los listeners globales

    return () => {

      document.removeEventListener("mouseout", handleMouseOut);

  }, []); // Solo se ejecuta una vez al montar

export function slugify(text) {
  return text
    .toString() // Asegura que el input sea una cadena de texto (por si viene un nÃºmero, null, etc.)

    .toLowerCase() // Convierte todo el texto a minÃºsculas
    // Ej: "MÃ¡rmol Blanco" â†’ "mÃ¡rmol blanco"

    .normalize("NFD") // Descompone caracteres acentuados (Ã‘ â†’ N + ~), necesario para quitar tildes
    // Ej: "mÃ¡rmol" â†’ "maÌrmol" (la "Ã¡" se convierte en "a" + acento)

    .replace(/[\u0300-\u036f]/g, "") // Elimina cualquier marca de acento (diacrÃ­ticos Unicode)
    // Ej: "maÌrmol" â†’ "marmol"

    .replace(/\s+/g, "-") // Reemplaza todos los espacios (y tabs) por guiones medios
    // Ej: "marmol blanco" â†’ "marmol-blanco"

    .replace(/[^\w\-]+/g, "") // Elimina cualquier caracter que no sea alfanumÃ©rico o guiÃ³n
    // Ej: "marmol-blanco@" â†’ "marmol-blanco"

    .replace(/\-\-+/g, "-") // Reemplaza mÃºltiples guiones seguidos por uno solo
    // Ej: "marmol--blanco" â†’ "marmol-blanco"

    .replace(/^-+|-+$/g, ""); // Elimina guiones al principio o al final de la cadena
    // Ej: "-marmol-blanco-" â†’ "marmol-blanco"
}
